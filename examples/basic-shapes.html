코드 예제

// Modern Vector.js를 사용한 기본 도형 그리기 예제

// 엔진 초기화
import { VectorEngine } from 'modern-vector.js';
import { ShapePlugin } from 'modern-vector.js/plugins/shapes';
import { CanvasRenderer } from 'modern-vector.js/plugins/renderers/canvas';

// 엔진 생성
const engine = new VectorEngine();

// 플러그인 등록
engine.use(new ShapePlugin());

// 렌더러 등록
const renderer = new CanvasRenderer({
  context: {
    canvas: document.getElementById('canvas')
  }
});
engine.renderer.register(renderer);
engine.renderer.setActive('canvas');

// 장면 생성
const scene = engine.scene.create();

// Rectangle 생성
const shapePlugin = engine.getPlugin('shape');
const rectangle = shapePlugin.createShape('rectangle', {
  x: 100,
  y: 100,
  width: 200,
  height: 100,
  style: {
    fillColor: '#3498db',
    strokeColor: '#2c3e50',
    strokeWidth: 2
  }
});

// 장면에 Shape 추가
scene.root.appendChild(rectangle);

// 렌더링
engine.renderer.render(scene);
