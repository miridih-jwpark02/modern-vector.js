function customDraw() {
  const ctx = rendererService.getContext();
  
  // 상태 저장
  ctx.save();
  
  // 그리기 설정
  ctx.fillStyle = 'red';
  ctx.strokeStyle = 'blue';
  ctx.lineWidth = 2;
  
  // 도형 그리기
  ctx.beginPath();
  ctx.moveTo(100, 100);
  ctx.lineTo(200, 100);
  ctx.lineTo(200, 200);
  ctx.lineTo(100, 200);
  ctx.closePath();
  ctx.fill();
  ctx.stroke();
  
  // 상태 복원
  ctx.restore();
}
